\section{Divisibility, GCD}


\begin{definition}[Divisibility]
\begin{gather*}
    \forall a,b \in \Z, \ b \ne 0,\ \exists c\in \Z \\  \ifthen{a=bc}{b \ \textbf{divides} \ a}
\end{gather*}
also written \[\divides{b}{a}\]
and if $b$ \underline{doesn't} divide $a$, \[\ndivides{b}{a}\]
\end{definition}
